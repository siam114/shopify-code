<script>
  (function() {
    class ReviewCarousel{{ ai_gen_id }} extends HTMLElement {
      constructor() {
        super();
        this.totalCards = 10;


        // Start on 5th review (index 4) — but position it in the middle “set” for seamless looping
        this.currentIndex = 4; // logical index (0..9) for dots
        this.positionIndex = this.totalCards + this.currentIndex; // physical index (10..19) in 3x track


        this.cardWidth = {{ block.settings.card_width }};
        this.cardGap = {{ block.settings.card_gap }};
        this.mobileCardWidth = 280;


        this._isJumping = false;
      }


      connectedCallback() {
        this.track = this.querySelector('.ai-review-carousel__track-{{ ai_gen_id }}');
        this.prevButton = this.querySelector('.ai-review-carousel__prev-{{ ai_gen_id }}');
        this.nextButton = this.querySelector('.ai-review-carousel__next-{{ ai_gen_id }}');
        this.dotsContainer = this.querySelector('.ai-review-carousel__dots-{{ ai_gen_id }}');


        this.cloneCards3x();
        this.createDots();
        this.bindEvents();


        // Initial render (no animation)
        this.jumpToPosition(this.positionIndex);


        window.addEventListener('resize', () => this.updateCarousel(false));
      }


      cloneCards3x() {
        const originals = Array.from(this.track.children);


        // rebuild: [clone set] + [original set] + [clone set]
        this.track.innerHTML = '';


        const addSet = (sourceNodes) => {
          sourceNodes.forEach(node => {
            const clone = node.cloneNode(true);
            this.track.appendChild(clone);
          });
        };


        addSet(originals); // set 0
        addSet(originals); // set 1 (middle)
        addSet(originals); // set 2


        // Cache all cards in DOM order — center styling is based on DOM index (positionIndex)
        this.allCards = Array.from(this.track.querySelectorAll('.ai-review-carousel__card-{{ ai_gen_id }}'));


        // Wrap logic: when crossing into outer sets, jump back to middle set (invisible)
        this.track.addEventListener('transitionend', () => {
          if (this._isJumping) return;


          if (this.positionIndex >= this.totalCards * 2) {
            this.positionIndex -= this.totalCards;
            this.jumpToPosition(this.positionIndex);
          }


          if (this.positionIndex < this.totalCards) {
            this.positionIndex += this.totalCards;
            this.jumpToPosition(this.positionIndex);
          }
        });
      }


      createDots() {
        this.dotsContainer.innerHTML = '';
        for (let i = 0; i < this.totalCards; i++) {
          const dot = document.createElement('button');
          dot.classList.add('ai-review-carousel__dot-{{ ai_gen_id }}');
          dot.type = 'button';
          dot.setAttribute('aria-label', `Go to review ${i + 1}`);
          if (i === this.currentIndex) dot.classList.add('active');
          dot.addEventListener('click', () => this.goToSlide(i));
          this.dotsContainer.appendChild(dot);
        }
      }


      bindEvents() {
        if (this.prevButton) this.prevButton.addEventListener('click', () => this.prev());
        if (this.nextButton) this.nextButton.addEventListener('click', () => this.next());


        // Touch swipe
        let startX = 0;
        let isDragging = false;


        this.track.addEventListener('touchstart', (e) => {
          startX = e.touches[0].clientX;
          isDragging = true;
        }, { passive: true });


        this.track.addEventListener('touchend', (e) => {
          if (!isDragging) return;
          const endX = e.changedTouches[0].clientX;
          const diff = startX - endX;


          if (Math.abs(diff) > 50) {
            if (diff > 0) this.next();
            else this.prev();
          }
          isDragging = false;
        }, { passive: true });
      }


      prev() {
        this.positionIndex -= 1;
        this.currentIndex = (this.currentIndex - 1 + this.totalCards) % this.totalCards;
        this.updateCarousel(true);
      }


      next() {
        this.positionIndex += 1;
        this.currentIndex = (this.currentIndex + 1) % this.totalCards;
        this.updateCarousel(true);
      }


      goToSlide(index) {
        this.currentIndex = index;
        // Always target the middle set so dots/jumps stay stable
        this.positionIndex = this.totalCards + index;
        this.updateCarousel(true);
      }


      getStep() {
        const viewportWidth = window.innerWidth;
        const isMobile = viewportWidth <= 749;
        const currentCardWidth = isMobile ? this.mobileCardWidth : this.cardWidth;
        return { viewportWidth, currentCardWidth, step: currentCardWidth + this.cardGap };
      }


      jumpToPosition(posIndex) {
        this._isJumping = true;
        this.track.style.transition = 'none';
        this.positionIndex = posIndex;
        this.applyTransform();
        void this.track.offsetHeight;
        this.track.style.transition = 'transform 0.6s cubic-bezier(0.4, 0, 0.2, 1)';
        this._isJumping = false;


        this.applyCenterClass();
        this.updateDots();
      }


      updateCarousel(animate) {
        this.track.style.transition = animate
          ? 'transform 0.6s cubic-bezier(0.4, 0, 0.2, 1)'
          : 'none';


        this.applyTransform();
        this.applyCenterClass();
        this.updateDots();


        if (!animate) {
          void this.track.offsetHeight;
          this.track.style.transition = 'transform 0.6s cubic-bezier(0.4, 0, 0.2, 1)';
        }
      }


      applyTransform() {
        const { viewportWidth, currentCardWidth, step } = this.getStep();
        const centerOffset = (viewportWidth / 2) - (currentCardWidth / 2);
        const offset = centerOffset - (this.positionIndex * step);
        this.track.style.transform = `translateX(${offset}px)`;
      }


      applyCenterClass() {
        // ONLY the physical center card (DOM index === positionIndex) becomes "center"
        this.allCards.forEach((card, idx) => {
          card.classList.toggle('center', idx === this.positionIndex);
        });
      }


      updateDots() {
        const dots = this.dotsContainer.querySelectorAll('.ai-review-carousel__dot-{{ ai_gen_id }}');
        dots.forEach((dot, i) => dot.classList.toggle('active', i === this.currentIndex));
      }
    }


    customElements.define('review-carousel-{{ ai_gen_id }}', ReviewCarousel{{ ai_gen_id }});
  })();
</script>
