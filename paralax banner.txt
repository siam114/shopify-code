<script>
document.addEventListener("DOMContentLoaded", function() {
  const parallax = document.getElementById("parallax-{{ section.id }}");


  const desktopBg = "{{ section.settings.bg_image | image_url: width: 2000 }}";
  const mobileBg = `{% if section.settings.mobile_bg_image != blank %}{{ section.settings.mobile_bg_image | image_url: width: 1000 }}{% else %}{{ section.settings.bg_image | image_url: width: 1000 }}{% endif %}`;


  function updateParallax() {
    if (window.innerWidth <= 768) {
      parallax.style.backgroundImage = `url('${mobileBg}')`;
    } else {
      parallax.style.backgroundImage = `url('${desktopBg}')`;
    }
    parallax.style.backgroundAttachment = "fixed";
    parallax.style.backgroundPosition = "center";
    parallax.style.backgroundSize = "cover";
  }


  updateParallax();
  window.addEventListener("resize", updateParallax);
});
</script>
