<script>
document.addEventListener("DOMContentLoaded", () => {
  const section = document.querySelector('#ella-js-sticky-{{ section.id }}');
  if (!section) return;


  const stickyBox = section.querySelector('#ellaStickyBox-{{ section.id }}');
  const visualCol = section.querySelector('.ella-js-visual-col');
  const mainImg = section.querySelector('#ellaMainImg-{{ section.id }}');
  const items = section.querySelectorAll('.ella-js-item');


  // Intersection Observer for Image Swapping
  const observer = new IntersectionObserver((entries) => {
    entries.forEach(entry => {
      if (entry.isIntersecting) {
        items.forEach(i => i.classList.remove('is-active'));
        entry.target.classList.add('is-active');
        const newSrc = entry.target.getAttribute('data-img-src');
        if (newSrc && mainImg.src !== newSrc) {
          mainImg.style.opacity = '0.4';
          setTimeout(() => { mainImg.src = newSrc; mainImg.style.opacity = '1'; }, 200);
        }
      }
    });
  }, { rootMargin: "-30% 0px -30% 0px", threshold: 0.1 });


  items.forEach(item => observer.observe(item));


  // JS Sticky Scroll Logic
  const handleSticky = () => {
    if (window.innerWidth < 992) return;


    const sectionRect = section.getBoundingClientRect();
    const colWidth = visualCol.offsetWidth;
    const stickyHeight = stickyBox.offsetHeight;
    const topOffset = 100;


    if (sectionRect.top <= topOffset) {
      if (sectionRect.bottom <= (stickyHeight + topOffset)) {
        stickyBox.classList.remove('fixed');
        stickyBox.classList.add('absolute-bottom');
        stickyBox.style.width = `100%`;
      } else {
        stickyBox.classList.remove('absolute-bottom');
        stickyBox.classList.add('fixed');
        stickyBox.style.width = `${colWidth}px`;
      }
    } else {
      stickyBox.classList.remove('fixed', 'absolute-bottom');
      stickyBox.style.width = `100%`;
    }
  };


  window.addEventListener('scroll', handleSticky);
  window.addEventListener('resize', handleSticky);
});
</script>
